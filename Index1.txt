// ===== INACTIVIDAD =====
const INACTIVITY_MS = 2 * 60 * 1000; // 2 min

function resetInactivityTimer() {
  clearTimeout(inactivityTimer);
  inactivityTimer = setTimeout(showInactivityWarning, INACTIVITY_MS);
}

function showInactivityWarning() {
  countdownVal = 30;
  document.getElementById('countdown-num').textContent = countdownVal;
  document.getElementById('inactivity-overlay').classList.add('show');
  clearInterval(countdownTimer);
  countdownTimer = setInterval(() => {
    countdownVal--;
    document.getElementById('countdown-num').textContent = countdownVal;
    if (countdownVal <= 0) {
      clearInterval(countdownTimer);
      resetApp();
    }
  }, 1000);
}

function dismissInactivity() {
  document.getElementById('inactivity-overlay').classList.remove('show');
  clearInterval(countdownTimer);
  resetInactivityTimer();
}

function resetApp() {
  document.getElementById('inactivity-overlay').classList.remove('show');
  clearInterval(countdownTimer);
  clearTimeout(inactivityTimer);
  document.getElementById('messages').innerHTML = '<div class="date-sep">Hoy</div>';
  document.getElementById('options-area').innerHTML = '';
  document.getElementById('input-area-container').innerHTML = '';
  initState();
  setTimeout(() => startFlow(), 400);
}

// ===== MENSAJES =====
function addMsg(html, type = 'bot', delay = 0) {
  return new Promise(resolve => {
    setTimeout(() => {
      const msgs = document.getElementById('messages');
      const div = document.createElement('div');
      div.className = `msg ${type}`;

      if (type === 'bot') {
        div.innerHTML = `<div class="avatar">☀️</div><div class="bubble">${html}</div>`;
      } else {
        div.innerHTML = `<div class="bubble">${html}</div>`;
      }

      msgs.appendChild(div);
      msgs.scrollTop = msgs.scrollHeight;
      resolve();
    }, delay);
  });
}

function showTyping(ms = 900) {
  return new Promise(resolve => {
    const msgs = document.getElementById('messages');
    const div = document.createElement('div');
    div.className = 'msg bot';
    div.id = 'typing-bubble';
    div.innerHTML = `<div class="avatar">☀️</div><div class="typing-indicator"><div class="typing-dot"></div><div class="typing-dot"></div><div class="typing-dot"></div></div>`;
    msgs.appendChild(div);
    msgs.scrollTop = msgs.scrollHeight;
    setTimeout(() => {
      const el = document.getElementById('typing-bubble');
      if (el) el.remove();
      resolve();
    }, ms);
  });
}

function clearOptions() {
  document.getElementById('options-area').innerHTML = '';
  document.getElementById('input-area-container').innerHTML = '';
}

function showOptions(opts) {
  const area = document.getElementById('options-area');
  area.innerHTML = '';
  opts.forEach(opt => {
    const btn = document.createElement('button');
    btn.className = `opt-btn${opt.primary ? ' primary' : ''}${opt.amarillo ? ' amarillo' : ''}`;
    btn.innerHTML = `${opt.icon ? `<span class="icon">${opt.icon}</span>` : ''}${opt.label}`;
    btn.addEventListener('click', () => {
      resetInactivityTimer();
      clearOptions();
      if (opt.value !== undefined) addMsg(opt.label, 'user');
      if (opt.action) opt.action(opt.value !== undefined ? opt.value : opt.label);
    });
    area.appendChild(btn);
  });
}

function showNumericInput(label, placeholder, unit, onSubmit, min = 0) {
  const container = document.getElementById('input-area-container');
  container.innerHTML = `
    <div class="input-area">
      <div class="input-wrap">
        <span class="input-label">${label}</span>
        <input type="number" class="num-input" id="num-field" placeholder="${placeholder}" min="${min}" inputmode="numeric">
      </div>
      <button class="send-btn" id="send-num">Continuar →</button>
    </div>
  `;

  const input = document.getElementById('num-field');
  const btn = document.getElementById('send-num');

  function submit() {
    const val = parseFloat(input.value);
    if (isNaN(val) || val < min) {
      input.style.borderColor = '#ef4444';
      setTimeout(() => input.style.borderColor = '', 1000);
      return;
    }
    resetInactivityTimer();
    clearOptions();
    addMsg(`${val} ${unit}`, 'user');
    onSubmit(val);
  }

  btn.addEventListener('click', submit);
  input.addEventListener('keydown', e => { if (e.key === 'Enter') submit(); });
  setTimeout(() => input.focus(), 100);
}

function showTextInput(label, placeholder, onSubmit) {
  const container = document.getElementById('input-area-container');
  container.innerHTML = `
    <div class="input-area">
      <div class="input-wrap">
        <span class="input-label">${label}</span>
        <input type="text" class="num-input" id="text-field" placeholder="${placeholder}" style="font-size:18px;font-weight:400;">
      </div>
      <button class="send-btn" id="send-txt">Continuar →</button>
    </div>
  `;

  const input = document.getElementById('text-field');
  const btn = document.getElementById('send-txt');

  function submit() {
    const val = input.value.trim();
    if (!val) return;
    resetInactivityTimer();
    clearOptions();
    addMsg(val, 'user');
    onSubmit(val);
  }

  btn.addEventListener('click', submit);
  input.addEventListener('keydown', e => { if (e.key === 'Enter') submit(); });
  setTimeout(() => input.focus(), 100);
}

];

function getSegmento(id) {
  return SEGMENTOS.find(s => s.id === id);
}

function calcAhorro(segId, gasolina) {
  const seg = getSegmento(segId);
  // Ahorro eléctrico: ~70% de la factura de referencia del segmento
  const ahorroElec = Math.round(seg.consumo * 0.70);
  // Ahorro gasolina: ~60% al combinar EV + solar
  const ahorroGaso = Math.round(gasolina * 0.60);
  const ahorroTotal = ahorroElec + ahorroGaso;
  const ahorroAnual = ahorroTotal * 12;
  const paybackAnios = (seg.costo / ahorroAnual).toFixed(1);

  return {
    seg,
    ahorroElec,
    ahorroGaso,
    ahorroTotal,
    ahorroAnual,
    kwp: seg.kwp,
    potencia: seg.potencia,
    costoSistema: seg.costo,
    costoKw: seg.costoKw,
    paybackAnios
  };
}
